make_midpoints <- function(vec,
                           breaks = seq(from = -7, to = 7, by = .5),
                           midpts = seq(from = -6.75, to = 6.75, by = .5)) {
  vec_cut <- cut(vec, breaks)
  midpts[as.numeric(vec_cut)]
}
